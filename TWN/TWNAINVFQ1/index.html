    Abstract

    • Data Source: Market Observation Post System (MOPS), Taiwan
    • Reference Code Table: TWN/ANPRCSTD
    • Code Column: coid

    import tejapi

    library("Tejapi")

yourdata<- Tejapi('TWN/AINVFQ1')

var result = await client.GetAsync("TWN/AINVFQ1");
